const fmtPrice = (w) => {
  const unit = w.priceUnit || "per job";
  if (w.priceType === "range" && w.rateMin && w.rateMax)
    return `₦${fmt(w.rateMin)} – ₦${fmt(w.rateMax)} ${unit}`;

  return `₦${fmt(w.rate)} ${unit}`;
};

// ─── PAYSTACK ────────────────────────────────────────────────
function openPaystack({ email, amountNaira, metadata, onSuccess, onClose }) {
  if (typeof window.PaystackPop === "undefined") {
    alert("Add Paystack script to public/index.html");
    return;
  }
  const handler = window.PaystackPop.setup({
    key: PAYSTACK_KEY, email, amount: amountNaira * 100, currency: "NGN",
    ref: "NS_" + Date.now(), metadata,
    callback: res => onSuccess(res),
    onClose,
  });
  handler.openIframe();
}

// ═══════════════════════════════════════════════════════════════
//  QUOTE THREAD MODAL  (chat + quote negotiation)
// ═══════════════════════════════════════════════════════════════
function QuoteThreadModal({ quote, isWorker, onClose, onUpdate }) {
  const [msg, setMsg]           = useState("");
  const [offerAmt, setOfferAmt] = useState("");
  const [showOffer, setShowOffer] = useState(false);
  const [paying, setPaying]     = useState(false);

  const sendMsg = () => {
    if (!msg.trim()) return;
    const updated = { ...quote, messages: [...quote.messages, { from: isWorker?"worker":"customer", text: msg.trim(), time: new Date().toLocaleTimeString([],{hour:"2-digit",minute:"2-digit"}) }] };
    onUpdate(updated); setMsg("");
  };

  const sendQuote = () => {
    const amt = parseInt(offerAmt);
    if (!amt || amt < 1) return;
    const label = isWorker ? "worker" : "customer";
    const text = isWorker
      ? `My quote for this job: ₦${fmt(amt)} ${quote.unit}. Please review and let me know if you'd like to proceed.`
      : `I'd like to offer ₦${fmt(amt)} ${quote.unit} for this job.`;
    const updated = { ...quote,
      workerPrice: isWorker ? amt : quote.workerPrice,
      status: "countered",
      messages: [...quote.messages, { from: label, text, time: new Date().toLocaleTimeString([],{hour:"2-digit",minute:"2-digit"}), isQuote:true, amount:amt }]
    };
    onUpdate(updated); setOfferAmt(""); setShowOffer(false);
  };

  const acceptQuote = (amt) => {
    const updated = { ...quote, agreedPrice: amt, status:"accepted",
      messages: [...quote.messages, { from: isWorker?"worker":"customer", text:`✅ Quote accepted! Agreed price: ₦${fmt(amt)} ${quote.unit}. Proceeding to payment.`, time: new Date().toLocaleTimeString([],{hour:"2-digit",minute:"2-digit"}), isSystem:true }]
    };
    onUpdate(updated);
  };

  const declineQuote = () => {
    const updated = { ...quote, status:"declined",
      messages: [...quote.messages, { from: isWorker?"worker":"customer", text:"❌ Quote declined.", time: new Date().toLocaleTimeString([],{hour:"2-digit",minute:"2-digit"}), isSystem:true }]
    };
    onUpdate(updated);
  };

  const handlePayment = () => {
    if (!quote.agreedPrice) return;
    const fee = Math.round(quote.agreedPrice * PLATFORM_FEE_PCT / 100);
    openPaystack({
      email: quote.customerEmail || "customer@naijaskills.com",
      amountNaira: fee,
      metadata: { type:"contact_fee", worker: quote.workerName, job: quote.jobTitle, agreed_price: quote.agreedPrice, fee },
      onSuccess: res => {
        const updated = { ...quote, status:"paid", payRef: res.reference,
          messages: [...quote.messages, { from:"system", text:`💳 Payment confirmed! ₦${fmt(fee)} platform fee paid. Ref: ${res.reference}. Worker contact: ${quote.workerPhone}`, time: new Date().toLocaleTimeString([],{hour:"2-digit",minute:"2-digit"}), isSystem:true }]
        };
        onUpdate(updated); setPaying(false);
      },
      onClose: () => setPaying(false),
    });
    setPaying(true);
  };

  const statusColors = { pending:{bg:"rgba(245,158,11,.1)",color:"#92400E"}, accepted:{bg:"rgba(0,135,81,.1)",color:G.green}, declined:{bg:"rgba(220,38,38,.1)",color:G.red}, countered:{bg:"rgba(37,99,235,.1)",color:G.blue}, paid:{bg:"rgba(0,135,81,.15)",color:G.greenDark} };
  const sc = statusColors[quote.status] || statusColors.pending;

  const latestQuoteMsg = [...quote.messages].reverse().find(m => m.isQuote);

  return (
    <div style={{position:"fixed",inset:0,background:"rgba(0,0,0,.55)",zIndex:350,display:"flex",alignItems:"flex-end",justifyContent:"center"}} onClick={e=>e.target===e.currentTarget&&onClose()}>
      <div className="slide-up" style={{background:"#fff",borderRadius:"24px 24px 0 0",width:"100%",maxWidth:520,height:"88vh",display:"flex",flexDirection:"column",position:"relative"}}>

        {/* Header */}
        <div style={{padding:"16px 18px 12px",borderBottom:"1px solid #E5E7EB",flexShrink:0}}>
          <button onClick={onClose} style={{position:"absolute",top:14,right:18,background:G.light,border:"none",borderRadius:"50%",width:30,height:30,display:"flex",alignItems:"center",justifyContent:"center",color:G.gray}}>✕</button>
          <div style={{fontFamily:"Syne,sans-serif",fontWeight:800,fontSize:"0.95rem",paddingRight:36,whiteSpace:"nowrap",overflow:"hidden",textOverflow:"ellipsis"}}>{quote.jobTitle}</div>
          <div style={{display:"flex",alignItems:"center",gap:8,marginTop:4,flexWrap:"wrap"}}>
            <span style={{fontSize:"0.75rem",color:G.gray}}>{isWorker ? quote.customerName : quote.workerName} · {quote.createdAt}</span>
            <span style={{background:sc.bg,color:sc.color,padding:"2px 10px",borderRadius:50,fontSize:"0.68rem",fontWeight:700,fontFamily:"Syne,sans-serif"}}>{QUOTE_STATUSES[quote.status]}</span>
            {quote.agreedPrice && <span style={{background:"rgba(0,135,81,.1)",color:G.green,padding:"2px 10px",borderRadius:50,fontSize:"0.68rem",fontWeight:700}}>Agreed: ₦{fmt(quote.agreedPrice)}</span>}
          </div>
        </div>

        {/* Job summary */}
        <div style={{padding:"10px 18px",background:"#F9FAFB",borderBottom:"1px solid #E5E7EB",flexShrink:0}}>
          <div style={{fontSize:"0.75rem",color:G.gray,display:"flex",gap:14,flexWrap:"wrap"}}>
            {quote.jobDate && <span>📅 {quote.jobDate}</span>}
            {quote.jobLocation && <span>📍 {quote.jobLocation}</span>}
            <span>💰 Worker rate: {fmtPrice({rate:quote.workerPrice||0,priceType:quote.workerPrice?"fixed":"negotiable",priceUnit:quote.unit})}</span>
          </div>
        </div>

        {/* Messages */}
        <div style={{flex:1,overflowY:"auto",padding:"14px 16px",display:"flex",flexDirection:"column",gap:10}}>
          {quote.messages.map((m, i) => {
            const isMe = (isWorker && m.from==="worker") || (!isWorker && m.from==="customer");
            const isSystem = m.isSystem || m.from==="system";
            if (isSystem) return (
              <div key={i} className="msg-in" style={{textAlign:"center",fontSize:"0.75rem",color:G.gray,background:"#F3F4F6",borderRadius:8,padding:"6px 12px",margin:"4px 20px"}}>{m.text}</div>
            );
            return (
              <div key={i} className="msg-in" style={{display:"flex",flexDirection:"column",alignItems:isMe?"flex-end":"flex-start"}}>
                {!isMe && <div style={{fontSize:"0.65rem",color:G.gray,marginBottom:2,marginLeft:4}}>{m.from==="worker"?quote.workerName:quote.customerName}</div>}
                <div style={{maxWidth:"78%",background:m.isQuote?(isMe?"linear-gradient(135deg,#008751,#00b86a)":"linear-gradient(135deg,#1E40AF,#2563EB)"):(isMe?"#111827":"#F3F4F6"),color:m.isQuote?"#fff":(isMe?"#fff":G.dark),borderRadius:isMe?"16px 16px 4px 16px":"16px 16px 16px 4px",padding:"10px 14px",fontSize:"0.84rem",lineHeight:1.5}}>
                  {m.isQuote && <div style={{fontFamily:"Syne,sans-serif",fontWeight:800,fontSize:"1rem",marginBottom:4}}>💰 ₦{fmt(m.amount)} {quote.unit}</div>}
                  {m.text}
                  {/* Accept/decline buttons for received quote */}
                  {m.isQuote && !isMe && quote.status!=="accepted" && quote.status!=="paid" && quote.status!=="declined" && (
                    <div style={{display:"flex",gap:6,marginTop:10}}>
                      <button onClick={()=>acceptQuote(m.amount)} style={{flex:1,background:"rgba(255,255,255,.2)",border:"1.5px solid rgba(255,255,255,.5)",color:"#fff",borderRadius:8,padding:"6px 0",fontSize:"0.75rem",fontFamily:"Syne,sans-serif",fontWeight:700}}>✅ Accept</button>
                      <button onClick={()=>declineQuote()} style={{flex:1,background:"rgba(255,255,255,.1)",border:"1.5px solid rgba(255,255,255,.3)",color:"rgba(255,255,255,.8)",borderRadius:8,padding:"6px 0",fontSize:"0.75rem"}}>❌ Decline</button>
                      <button onClick={()=>setShowOffer(true)} style={{flex:1,background:"rgba(255,255,255,.1)",border:"1.5px solid rgba(255,255,255,.3)",color:"rgba(255,255,255,.8)",borderRadius:8,padding:"6px 0",fontSize:"0.75rem"}}>🔄 Counter</button>
                    </div>
                  )}
                </div>
                <div style={{fontSize:"0.62rem",color:G.gray,marginTop:2,marginLeft:4,marginRight:4}}>{m.time}</div>
              </div>
            );
          })}
        </div>

        {/* Accepted — pay now */}
        {quote.status==="accepted" && (

function QuotesPage({ quotes, onUpdateQuote }) {
  const [view, setView]         = useState("customer"); // "customer" | "worker"
  const [openQuote, setOpenQuote] = useState(null);
  const [search, setSearch]     = useState("");

  const filtered = quotes.filter(q => {
    const matchView = view==="worker" ? true : true; // in real app filter by logged-in user
    const q2 = search.toLowerCase();

//  ROOT APP

export default function App() {
  const [page,setPage]       = useState("directory");
  const [workers,setWorkers] = useState(INITIAL_WORKERS);
  const [quotes,setQuotes]   = useState(INITIAL_QUOTES);
  const addQuote    = q   => setQuotes(qs=>[q,...qs]);
  const updateQuote = q   => setQuotes(qs=>qs.map(x=>x.id===q.id?q:x));

  useEffect(()=>{
    const params = new URLSearchParams(window.location.search);
    if(params.get("trusted")===TRUSTED_SECRET) setPage("trusted-private");
  },[]);

  const addWorker    = w      => setWorkers(ws=>[...ws,w]);
  const updateWorker = (id,d) => setWorkers(ws=>ws.map(w=>w.id===id?{...w,...d}:w));
  const deleteWorker = id     => setWorkers(ws=>ws.filter(w=>w.id!==id||w.trusted));
  const addTrusted   = w      => setWorkers(ws=>[...ws,{...w,trusted:true,status:"approved",verified:true,regPaid:true,regExpiry:"N/A"}]);

  return (
    <>
      <style>{css}</style>
      {page!=="trusted-private"&&<Header page={page} setPage={setPage}/>}
      {page==="directory"      &&<DirectoryPage workers={workers} onAddQuote={addQuote}/>}
      {page==="register"       &&<RegisterPage onRegister={addWorker}/>}
      {page==="quotes"         &&<QuotesPage quotes={quotes} onUpdateQuote={updateQuote}/>}
      {page==="admin"          &&<AdminPage workers={workers} onUpdateWorker={updateWorker} onDeleteWorker={deleteWorker} onAddTrusted={addTrusted}/>}
      {page==="trusted-private"&&<TrustedPrivatePage onAdd={addTrusted}/>}
      {page!=="trusted-private"&&(
        <footer style={{textAlign:"center",padding:"22px 18px",fontSize:"0.76rem",color:G.gray,borderTop:"1px solid #E5E7EB",marginTop:10}}>
          <strong style={{color:G.green,fontFamily:"Syne,sans-serif"}}>NaijaSkills</strong> — Connecting Nigerian talent to opportunity 🇳🇬
          <div style={{fontSize:"0.7rem",marginTop:4}}>₦{fmt(REG_FEE)}/year registration · 10% contact fee</div>
        </footer>
      )}
    </>
  );
}
